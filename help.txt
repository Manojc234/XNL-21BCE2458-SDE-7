const Sidebar = ({ role }) => {
  const [activeMenu, setActiveMenu] = useState(null);

  const handleHover = (menu) => {
    setActiveMenu(menu);
  };

  return (
    <div className="sidebar-container">
      {/* Left Blue Sidebar */}
      <div className="sidebar">
        <div className="sidebar-header">Dashboard</div>
        {menuData.map((menu, idx) => (
          <div
            key={idx}
            className="menu-item"
            onMouseEnter={() => handleHover(menu.title)}
            onMouseLeave={() => setActiveMenu(null)}
          >
            {menu.icon}
            <span>{menu.title}</span>
          </div>
        ))}
      </div>

      {/* Right White Submenu Panel */}
      <div className="submenu-panel">
        {menuData
          .filter((m) => m.title === activeMenu)
          .map((menu, i) => (
            <div key={i}>
              <h3 className="submenu-title">{menu.title}</h3>
              {menu.children &&
                menu.children.map((child, j) => (
                  <div key={j} className="submenu-section">
                    <strong>{child.title}</strong>
                    <ul>
                      {child.children &&
                        child.children.map((sub, k) => (
                          <li key={k}>{sub.title}</li>
                        ))}
                    </ul>
                  </div>
                ))}
            </div>
          ))}
      </div>
    </div>
  );
};

export default Sidebar;
